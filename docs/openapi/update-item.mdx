# Update item

<OperationHeader
  serverUrl="https://api-staging.xpayeg.com"
  method="PUT"
  path="/items/{itemId}"
  hasAuth
  hasPath
  hasBody
/>

Update an existing resource by ID.

## Authorization

<div className="openapi-section">

| Name | Type | In | Description |
|------|------|-----|-------------|
| Authorization | Bearer \<token\> | header | JWT or API token |

</div>

## Path Parameters

<div className="openapi-section">

| Name | Type | Required | Description |
|------|------|----------|-------------|
| itemId | string | Yes | The ID of the item to update |

</div>

## Request Body

<div className="openapi-section">

| Name | Type | Required | Description |
|------|------|----------|-------------|
| name | string | No | New display name |
| amount | integer | No | New amount in smallest currency unit |

```json
{
  "name": "Updated Item",
  "amount": 2000
}
```

</div>

const itemId = 'item_01HXYZ';
const response = await fetch(`https://api-staging.xpayeg.com/items/${itemId}`, {

  headers: {
    'Content-Type': 'application/json',
    Authorization: 'Bearer YOUR_ACCESS_TOKEN',
  },
  body: JSON.stringify({ name: 'Updated Item', amount: 2000 }),
});
const data = await response.json();

import requests

        "Authorization": "Bearer YOUR_ACCESS_TOKEN",
    },
    json={"name": "Updated Item", "amount": 2000},
